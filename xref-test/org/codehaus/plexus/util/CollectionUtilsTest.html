30      public void testMergeMaps()
31      {
32          Map<String,String> dominantMap = new HashMap<String,String>();
33          dominantMap.put( "a", "a" );
34          dominantMap.put( "b", "b" );
35          dominantMap.put( "c", "c" );
36          dominantMap.put( "d", "d" );
37          dominantMap.put( "e", "e" );
38          dominantMap.put( "f", "f" );
39  
40          Map<String,String> recessiveMap = new HashMap<String,String>();
41          recessiveMap.put( "a", "invalid" );
42          recessiveMap.put( "b", "invalid" );
43          recessiveMap.put( "c", "invalid" );
44          recessiveMap.put( "x", "x" );
45          recessiveMap.put( "y", "y" );
46          recessiveMap.put( "z", "z" );
47  
48          Map<String, String> result = CollectionUtils.mergeMaps( dominantMap, recessiveMap );
49  
50          // We should have 9 elements
51          assertEquals( 9, result.keySet().size() );
52  
53          // Check the elements.
54          assertEquals( "a", result.get( "a" ) );
55          assertEquals( "b", result.get( "b" ) );
56          assertEquals( "c", result.get( "c" ) );
57          assertEquals( "d", result.get( "d" ) );
58          assertEquals( "e", result.get( "e" ) );
59          assertEquals( "f", result.get( "f" ) );
60          assertEquals( "x", result.get( "x" ) );
61          assertEquals( "y", result.get( "y" ) );
62          assertEquals( "z", result.get( "z" ) );
63      }

69          Map<String, String> result0 = CollectionUtils.mergeMaps( new Map[]

78  
79          Map<String, String> result1 = CollectionUtils.mergeMaps( new Map[]
80          {
81              map1
82          } );
83  
84          assertEquals( "a", result1.get( "a" ) );
85  
86          // Test with an array with two elements.
87          Map<String, String>  map2 = new HashMap<String, String> ();
88          map2.put( "a", "aa" );
89          map2.put( "b", "bb" );
90  
91          Map<String, String>  result2 = CollectionUtils.mergeMaps( new Map[]
92          {
93              map1,
94              map2
95          } );
96  
97          assertEquals( "a", result2.get( "a" ) );
98          assertEquals( "bb", result2.get( "b" ) );
99  
100         // Now swap the dominant order.
101         Map<String, String>  result3 = CollectionUtils.mergeMaps( new Map[]
102         {
103             map2,
104             map1
105         } );
106 
107         assertEquals( "aa", result3.get( "a" ) );
108         assertEquals( "bb", result3.get( "b" ) );
109

140     public void testMavenPropertiesLoading()
141     {
142         // Mimic MavenSession properties loading. Properties listed
143         // in dominant order.
144         Properties systemProperties = new Properties();
145         Properties userBuildProperties = new Properties();
146         Properties projectBuildProperties = new Properties();
147         Properties projectProperties = new Properties();
148         Properties driverProperties = new Properties();
149 
150         // System properties
151         systemProperties.setProperty( "maven.home", "/projects/maven" );
152 
153         // User build properties
154         userBuildProperties.setProperty( "maven.username", "jvanzyl" );
155         userBuildProperties.setProperty( "maven.repo.remote.enabled", "false" );
156         userBuildProperties.setProperty( "maven.repo.local", "/opt/maven/artifact" );
157 
158         // Project build properties
159         projectBuildProperties.setProperty( "maven.final.name", "maven" );
160 
161         String mavenRepoRemote = "http://www.ibiblio.org/maven,http://foo/bar";
162 
163         // Project properties
164         projectProperties.setProperty( "maven.repo.remote", mavenRepoRemote );
165 
166         String basedir = "/home/jvanzyl/projects/maven";
167 
168         // Driver properties
169         driverProperties.setProperty( "basedir", basedir );
170         driverProperties.setProperty( "maven.build.src", "${basedir}/src" );
171         driverProperties.setProperty( "maven.build.dir", "${basedir}/target" );
172         driverProperties.setProperty( "maven.build.dest", "${maven.build.dir}/classes" );
173         driverProperties.setProperty( "maven.repo.remote", "http://www.ibiblio.org/maven" );
174         driverProperties.setProperty( "maven.final.name", "maven-1.0" );
175         driverProperties.setProperty( "maven.repo.remote.enabled", "true" );
176         driverProperties.setProperty( "maven.repo.local", "${maven.home}/artifact" );
177 
178         Map result = CollectionUtils.mergeMaps( new Map[]
179         {
180             systemProperties,
181             userBuildProperties,
182             projectBuildProperties,
183             projectProperties,
184             driverProperties
185         } );
186 
187         // Values that should be taken from systemProperties.
188         assertEquals( "/projects/maven", (String) result.get( "maven.home" ) );
189 
190         // Values that should be taken from userBuildProperties.
191         assertEquals( "/opt/maven/artifact", (String) result.get( "maven.repo.local" ) );
192         assertEquals( "false", (String) result.get( "maven.repo.remote.enabled" ) );
193         assertEquals( "jvanzyl", (String) result.get( "maven.username" ) );
194 
195         // Values take from projectBuildProperties.
196         assertEquals( "maven", (String) result.get( "maven.final.name" ) );
197 
198         // Values take from projectProperties.
199         assertEquals( mavenRepoRemote, (String) result.get( "maven.repo.remote" ) );
200     }
