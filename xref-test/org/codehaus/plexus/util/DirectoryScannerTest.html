146     /**
147      * Check if 'src/test/resources/symlinks/src/sym*' test files (start with 'sym') exist and are symlinks.<br>
148      * On some OS (like Windows 10), the 'git clone' requires to be executed with admin permissions and the
149      * 'core.symlinks=true' git option.
150      * 
151      * @return true If files here and symlinks, false otherwise
152      */
153     private boolean checkTestFilesSymlinks()
154     {
155         File symlinksDirectory = new File( "src/test/resources/symlinks/src" );
156         try
157         {
158             List<String> symlinks =
159                 FileUtils.getFileAndDirectoryNames( symlinksDirectory, "sym*", null, true, true, true, true );
160             if ( symlinks.isEmpty() )
161             {
162                 throw new IOException( "Symlinks files/directories are not present" );
163             }
164             for ( String symLink : symlinks )
165             {
166                 if ( !Files.isSymbolicLink( Paths.get( symLink ) ) )
167                 {
168                     throw new IOException( String.format( "Path is not a symlink: %s", symLink ) );
169                 }
170             }
171             return true;
172         }
173         catch ( IOException e )
174         {
175             System.err.println( String.format( "The unit test '%s.%s' will be skipped, reason: %s",
176                                                this.getClass().getSimpleName(), name.getMethodName(),
177                                                e.getMessage() ) );
178             System.out.println( String.format( "This test requires symlinks files in '%s' directory.",
179                                                symlinksDirectory.getPath() ) );
180             System.out.println( "On some OS (like Windows 10), files are present only if the clone/checkout is done"
181                 + " in administrator mode, and correct (symlinks and not flat file/directory)"
182                 + " if symlinks option are used (for git: git clone -c core.symlinks=true [url])" );
183             return false;
184         }
185     }

528             "directoryTest" + File.separator + "test-dir-123" + File.separator + "anotherDir3" + File.separator

531

533             "directoryTest" + File.separator + "testDir123" + File.separator + "file1.dat",
534             "directoryTest" + File.separator + "test_dir_123" + File.separator + "file1.dat",
535             "directoryTest" + File.separator + "test-dir-123" + File.separator + "file1.dat"

537 
538         final Set<String> scannedDirSet = new HashSet<String>();
539 
540         DirectoryScanner ds = new DirectoryScanner()
541         {

603     private void assertInclusionsAndExclusions( String[] files, String[] excludedPaths, String... includedPaths )
604     {
605         Arrays.sort( files );
606 
607         System.out.println( "Included files: " );
608         for ( String file : files )
609         {
610             System.out.println( file );
611         }
612 
613         List<String> failedToExclude = new ArrayList<String>();
614         for ( String excludedPath : excludedPaths )
615         {
616             String alt = excludedPath.replace( '/', '\\' );
617             System.out.println( "Searching for exclusion as: " + excludedPath + "\nor: " + alt );
618             if ( Arrays.binarySearch( files, excludedPath ) > -1 || Arrays.binarySearch( files, alt ) > -1 )
619             {
620                 failedToExclude.add( excludedPath );
621             }
622         }
623 
624         List<String> failedToInclude = new ArrayList<String>();
625         for ( String includedPath : includedPaths )
626         {
627             String alt = includedPath.replace( '/', '\\' );
628             System.out.println( "Searching for inclusion as: " + includedPath + "\nor: " + alt );
629             if ( Arrays.binarySearch( files, includedPath ) < 0 && Arrays.binarySearch( files, alt ) < 0 )
630             {
631                 failedToInclude.add( includedPath );
632             }
633         }
634 
635         StringBuilder buffer = new StringBuilder();
636         if ( !failedToExclude.isEmpty() )
637         {
638             buffer.append( "Should NOT have included:\n" ).append( StringUtils.join( failedToExclude.iterator(),
639                                                                                      "\n\t- " ) );
640         }
641 
642         if ( !failedToInclude.isEmpty() )
643         {
644             if ( buffer.length() > 0 )
645             {
646                 buffer.append( "\n\n" );
647             }
648 
649             buffer.append( "Should have included:\n" ).append( StringUtils.join( failedToInclude.iterator(),
650                                                                                  "\n\t- " ) );
651         }
652 
653         if ( buffer.length() > 0 )
654         {
655             fail( buffer.toString() );
656         }
657     }

659     private void createFiles( File dir, String... paths )
660         throws IOException
661     {
662         for ( String path1 : paths )
663         {
664             String path = path1.replace( '/', File.separatorChar ).replace( '\\', File.separatorChar );
665             File file = new File( dir, path );
666 
667             if ( path.endsWith( File.separator ) )
668             {
669                 file.mkdirs();
670             }
671             else
672             {
673                 if ( file.getParentFile() != null )
674                 {
675                     file.getParentFile().mkdirs();
676                 }
677 
678                 createFile( file, 0 );
679             }
680         }
681     }
