36  /**
37   * Base class for testcases doing tests with files.
38   *
39   * @author Jeremias Maerki
40   */
41  public abstract class FileBasedTestCase
42      extends TestCase
43  {
44      private static File testDir;
45  
46      public static File getTestDirectory()
47      {
48          if ( testDir == null )
49          {
50              testDir = ( new File( "target/test/io/" ) ).getAbsoluteFile();
51          }
52          return testDir;
53      }
54  
55      protected byte[] createFile( final File file, final long size )
56          throws IOException
57      {
58          if ( !file.getParentFile().exists() )
59          {
60              throw new IOException( "Cannot create file " + file + " as the parent directory does not exist" );
61          }
62  
63          byte[] data = generateTestData( size );
64  
65          final BufferedOutputStream output = new BufferedOutputStream( new FileOutputStream( file ) );
66  
67          try
68          {
69              output.write( data );
70  
71              return data;
72          }
73          finally
74          {
75              output.close();
76          }
77      }
78  
79      protected boolean createSymlink( final File link, final File target )
80      {
81          try
82          {
83              String[] args = { "ln", "-s", target.getAbsolutePath(), link.getAbsolutePath() };
84              Process process = Runtime.getRuntime().exec( args );
85              process.waitFor();
86              if ( 0 != process.exitValue() )
87              {
88                  return false;
89              }
90          }
91          catch ( Exception e )
92          {
93              // assume platform does not support "ln" command, tests should be skipped
94              return false;
95          }
96          return true;
97      }
98  
99      protected byte[] generateTestData( final long size )
100     {
101         try
102         {
103             ByteArrayOutputStream baout = new ByteArrayOutputStream();
104             generateTestData( baout, size );
105             return baout.toByteArray();
106         }
107         catch ( IOException ioe )
108         {
109             throw new RuntimeException( "This should never happen: " + ioe.getMessage() );
110         }
111     }
112 
113     protected void generateTestData( final OutputStream out, final long size )
114         throws IOException
115     {
116         for ( int i = 0; i < size; i++ )
117         {
118             //output.write((byte)'X');
119 
120             // nice varied byte pattern compatible with Readers and Writers
121             out.write( (byte) ( ( i % 127 ) + 1 ) );
122         }
123     }
124 
125     protected File newFile( String filename ) throws IOException
126     {
127         final File destination = new File( getTestDirectory(), filename );
128         /*
129         assertTrue( filename + "Test output data file shouldn't previously exist",
130                     !destination.exists() );
131         */
132         if ( destination.exists() )
133         {
134             FileUtils.forceDelete( destination );
135         }
136         return destination;
137     }
138 
139     protected void checkFile( final File file, final File referenceFile )
140         throws Exception
141     {
142         assertTrue( "Check existence of output file", file.exists() );
143         assertEqualContent( referenceFile, file );
144     }
145 
146     protected void checkWrite( final OutputStream output ) throws Exception
147     {
148         try
149         {
150             new PrintStream( output ).write( 0 );
151         }
152         catch ( final Throwable t )
153         {
154             throw new AssertionFailedError( "The copy() method closed the stream " + "when it shouldn't have. "
155                 + t.getMessage() );
156         }
157     }
158 
159     protected void checkWrite( final Writer output ) throws Exception
160     {
161         try
162         {
163             new PrintWriter( output ).write( 'a' );
164         }
165         catch ( final Throwable t )
166         {
167             throw new AssertionFailedError( "The copy() method closed the stream " + "when it shouldn't have. "
168                 + t.getMessage() );
169         }
170     }
171 
172     protected void deleteFile( final File file )
173         throws Exception
174     {
175         if ( file.exists() )
176         {
177             assertTrue( "Couldn't delete file: " + file, file.delete() );
178         }
179     }

185     /** Assert that the content of two files is the same. */
186     private void assertEqualContent( final File f0, final File f1 )
187         throws IOException
188     {
189         /* This doesn't work because the filesize isn't updated until the file
190          * is closed.
191         assertTrue( "The files " + f0 + " and " + f1 +
192                     " have differing file sizes (" + f0.length() +
193                     " vs " + f1.length() + ")", ( f0.length() == f1.length() ) );
194         */
195         final InputStream is0 = new FileInputStream( f0 );
196         try
197         {
198             final InputStream is1 = new FileInputStream( f1 );
199             try
200             {
201                 final byte[] buf0 = new byte[1024];
202                 final byte[] buf1 = new byte[1024];
203                 int n0 = 0;
204                 int n1 = 0;
205 
206                 while ( -1 != n0 )
207                 {
208                     n0 = is0.read( buf0 );
209                     n1 = is1.read( buf1 );
210                     assertTrue( "The files " + f0 + " and " + f1 +
211                                 " have differing number of bytes available (" + n0 +
212                                 " vs " + n1 + ")", ( n0 == n1 ) );
213 
214                     assertTrue( "The files " + f0 + " and " + f1 +
215                                 " have different content", Arrays.equals( buf0, buf1 ) );
216                 }
217             }
218             finally
219             {
220                 is1.close();
221             }
222         }
223         finally
224         {
225             is0.close();
226         }
227     }
228 
229     /** Assert that the content of a file is equal to that in a byte[]. */
230     protected void assertEqualContent( final byte[] b0, final File file )
231         throws IOException
232     {
233         final InputStream is = new FileInputStream( file );
234         try
235         {
236             byte[] b1 = new byte[b0.length];
237             int numRead = is.read( b1 );
238             assertTrue( "Different number of bytes", numRead == b0.length && is.available() == 0 );
239             for ( int i = 0;
240                   i < numRead;
241                   assertTrue( "Byte " + i + " differs (" + b0[i] + " != " + b1[i] + ")",
242                               b0[i] == b1[i] ), i++
243                 )
244                 ;
245         }
246         finally
247         {
248             is.close();
249         }
250     }
251 
252     protected void assertIsDirectory( File file )
253     {
254         assertTrue( "The File doesn't exists: " + file.getAbsolutePath(), file.exists() );
255 
256         assertTrue( "The File isn't a directory: " + file.getAbsolutePath(), file.isDirectory() );
257     }
258 
259     protected void assertIsFile( File file )
260     {
261         assertTrue( "The File doesn't exists: " + file.getAbsolutePath(), file.exists() );
262 
263         assertTrue( "The File isn't a file: " + file.getAbsolutePath(), file.isFile() );
264     }
