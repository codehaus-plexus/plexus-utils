37  /**
38   * This is used to test IOUtil for correctness. The following checks are performed:
39   * <ul>
40   *   <li>The return must not be null, must be the same type and equals() to the method's second arg</li>
41   *   <li>All bytes must have been read from the source (available() == 0)</li>
42   *   <li>The source and destination content must be identical (byte-wise comparison check)</li>
43   *   <li>The output stream must not have been closed (a byte/char is written to test this, and
44   *   subsequent size checked)</li>
45   * </ul>
46   * Due to interdependencies in IOUtils and IOUtilsTestlet, one bug may cause
47   * multiple tests to fail.
48   *
49   * @author <a href="mailto:jefft@apache.org">Jeff Turner</a>
50   */

52      extends TestCase

97      private void createFile( File file, long size )
98          throws IOException
99      {
100         BufferedOutputStream output =
101             new BufferedOutputStream( new FileOutputStream( file ) );
102 
103         for ( int i = 0; i < size; i++ )
104         {
105             output.write( (byte) ( i % 128 ) ); // nice varied byte pattern compatible with Readers and Writers
106         }
107 
108         output.close();
109     }

117     /** Assert that the content of two files is the same. */
118     private void assertEqualContent( File f0, File f1 )
119         throws IOException
120     {
121         FileInputStream is0 = new FileInputStream( f0 );
122         FileInputStream is1 = new FileInputStream( f1 );
123         byte[] buf0 = new byte[FILE_SIZE];
124         byte[] buf1 = new byte[FILE_SIZE];
125         int n0 = 0;
126         int n1 = 0;
127 
128         try
129         {
130             while ( 0 <= n0 )
131             {
132                 n0 = is0.read( buf0 );
133                 n1 = is1.read( buf1 );
134                 assertTrue( "The files " + f0 + " and " + f1 +
135                             " have differing number of bytes available (" + n0 +
136                             " vs " + n1 + ")", ( n0 == n1 ) );
137 
138                 assertTrue( "The files " + f0 + " and " + f1 +
139                             " have different content", Arrays.equals( buf0, buf1 ) );
140             }
141         }
142         finally
143         {
144             is0.close();
145             is1.close();
146         }
147     }

149     /** Assert that the content of a file is equal to that in a byte[]. */
150     private void assertEqualContent( byte[] b0, File file )
151         throws IOException
152     {
153         FileInputStream is = new FileInputStream( file );
154         byte[] b1 = new byte[b0.length];
155         int numRead = is.read( b1 );
156         assertTrue( "Different number of bytes", numRead == b0.length && is.available() == 0 );
157         for ( int i = 0;
158               i < numRead;
159               assertTrue( "Byte " + i + " differs (" + b0[i] + " != " + b1[i] + ")", b0[i] == b1[i] ), i++
160             )
161             ;
162         is.close();
163     }

165     public void testInputStreamToOutputStream()
166         throws Exception
167     {
168         File destination = newFile( "copy1.txt" );
169         FileInputStream fin = new FileInputStream( testFile );
170         FileOutputStream fout = new FileOutputStream( destination );
171 
172         IOUtil.copy( fin, fout );
173         assertTrue( "Not all bytes were read", fin.available() == 0 );
174         fout.flush();
175 
176         checkFile( destination );
177         checkWrite( fout );
178         fout.close();
179         fin.close();
180         deleteFile( destination );
181     }

227         // Note: rely on the method to flush

510     private File newFile( String filename )
511         throws Exception
512     {
513         File destination = new File( testDirectory, filename );
514         assertTrue( filename + "Test output data file shouldn't previously exist",
515                     !destination.exists() );
516 
517         return destination;

520     private void checkFile( File file )
521         throws Exception
522     {
523         assertTrue( "Check existence of output file", file.exists() );
524         assertEqualContent( testFile, file );
525     }
526 
527     private void checkWrite( OutputStream output )
528         throws Exception
529     {
530         try
531         {
532             new PrintStream( output ).write( 0 );
533         }
534         catch ( Throwable t )
535         {
536             throw new AssertionFailedError( "The copy() method closed the stream " +
537                                             "when it shouldn't have. " + t.getMessage() );
538         }
539     }
540 
541     private void checkWrite( Writer output )
542         throws Exception
543     {
544         try
545         {
546             new PrintWriter( output ).write( 'a' );
547         }
548         catch ( Throwable t )
549         {
550             throw new AssertionFailedError( "The copy() method closed the stream " +
551                                             "when it shouldn't have. " + t.getMessage() );
552         }
553     }

555     private void deleteFile( File file )
556         throws Exception
557     {
558         assertTrue( "Wrong output size: file.length()=" +
559                     file.length() + "!=" + FILE_SIZE + 1,
560                     file.length() == FILE_SIZE + 1 );
561 
562         assertTrue( "File would not delete", ( file.delete() || ( !file.exists() ) ) );
563     }
