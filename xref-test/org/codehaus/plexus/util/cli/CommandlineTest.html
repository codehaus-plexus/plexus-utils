91      public void testExecute()
92      {
93          try
94          {
95              // allow it to detect the proper shell here.
96              Commandline cmd = new Commandline();
97              cmd.setWorkingDirectory( baseDir );
98              cmd.setExecutable( "echo" );
99              assertEquals( "echo", cmd.getShell().getOriginalExecutable() );
100             cmd.createArgument().setValue( "Hello" );
101 
102             StringWriter swriter = new StringWriter();
103             Process process = cmd.execute();
104 
105             Reader reader = new InputStreamReader( process.getInputStream() );
106 
107             char[] chars = new char[16];
108             int read = -1;
109             while ( ( read = reader.read( chars ) ) > -1 )
110             {
111                 swriter.write( chars, 0, read );
112             }
113 
114             String output = swriter.toString().trim();
115 
116             assertEquals( "Hello", output );
117         }
118         catch ( Exception e )
119         {
120             fail( e.getMessage() );
121         }
122     }

535     /**
536      * Make the file executable for Unix box.
537      *
538      * @param path not null
539      * @throws IOException if any
540      */
541     private static void makeExecutable( File path )
542         throws IOException
543     {
544         if ( path == null )
545         {
546             throw new IllegalArgumentException( "The file is null" );
547         }
548 
549         if ( !path.isFile() )
550         {
551             throw new IllegalArgumentException( "The file '" + path.getAbsolutePath() + "' should be a file" );
552         }
553 
554         if ( !Os.isFamily( Os.FAMILY_WINDOWS ) )
555         {
556             Process proc = Runtime.getRuntime().exec( new String[] { "chmod", "a+x", path.getAbsolutePath() } );
557             while ( true )
558             {
559                 try
560                 {
561                     proc.waitFor();
562                     break;
563                 }
564                 catch ( InterruptedException e )
565                 {
566                     // ignore
567                 }
568             }
569         }
570     }

572     /**
573      * Create and execute a script file in the given dir with the given content. The script file will be called
574      * <code>echo.bat</code> for Windows box, otherwise <code>echo</code>.
575      *
576      * @param dir the parent dir where echo.bat or echo will be created
577      * @param content the content of the script file
578      * @throws Exception if any
579      */
580     private static void createAndCallScript( File dir, String content )
581         throws Exception
582     {
583         if ( !dir.exists() )
584         {
585             assertTrue( "Can't create dir:" + dir.getAbsolutePath(), dir.mkdirs() );
586         }
587 
588         // Create a script file
589         File bat;
590         if ( Os.isFamily( Os.FAMILY_WINDOWS ) )
591         {
592             bat = new File( dir, "echo.bat" );
593         }
594         else
595         {
596             bat = new File( dir, "echo" );
597         }
598 
599         Writer w = new FileWriter( bat );
600         try
601         {
602             IOUtil.copy( content, w );
603         }
604         finally
605         {
606             IOUtil.close( w );
607         }
608 
609         // Change permission
610         makeExecutable( bat );
611 
612         Commandline cmd = new Commandline();
613         cmd.setExecutable( bat.getAbsolutePath() );
614         cmd.setWorkingDirectory( dir );
615 
616         // Execute the script file
617         executeCommandLine( cmd );
618     }

620     /**
621      * Execute the command line
622      *
623      * @param cmd not null
624      * @throws Exception if any
625      */
626     private static void executeCommandLine( Commandline cmd )
627         throws Exception
628     {
629         CommandLineUtils.StringStreamConsumer err = new CommandLineUtils.StringStreamConsumer();
630 
631         try
632         {
633             System.out.println( "Command line is: " + StringUtils.join( cmd.getShellCommandline(), " " ) );
634 
635             int exitCode = CommandLineUtils.executeCommandLine( cmd, new DefaultConsumer(), err );
636 
637             if ( exitCode != 0 )
638             {
639                 String msg = "Exit code: " + exitCode + " - " + err.getOutput();
640                 throw new Exception( msg.toString() );
641             }
642         }
643         catch ( CommandLineException e )
644         {
645             throw new Exception( "Unable to execute command: " + e.getMessage(), e );
646         }
