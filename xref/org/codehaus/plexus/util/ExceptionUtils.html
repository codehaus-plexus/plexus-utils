92      protected static String[] CAUSE_METHOD_NAMES = {
93          "getCause",
94          "getNextException",

111     /**
112      * <p>Adds to the list of method names used in the search for <code>Throwable</code>
113      * objects.</p>
114      *
115      * @param methodName  the methodName to add to the list, null and empty strings are ignored
116      */
117     public static void addCauseMethodName( String methodName )
118     {
119         if ( methodName != null && methodName.length() > 0 )
120         {
121             List<String> list = new ArrayList<String>( Arrays.asList( CAUSE_METHOD_NAMES ) );
122             list.add( methodName );
123             CAUSE_METHOD_NAMES = list.toArray( new String[list.size()] );
124         }
125     }

127     /**
128      * <p>Introspects the specified <code>Throwable</code> to obtain the cause.</p>
129      *
130      * <p>The method searches for methods with specific names that return a
131      * <code>Throwable</code> object. This will pick up most wrapping exceptions,
132      * including those from JDK 1.4, and
133      * The method names can be added to using {@link #addCauseMethodName(String)}.
134      * The default list searched for are:</p>
135      * <ul>
136      * <li><code>getCause()</code>
137      * <li><code>getNextException()</code>
138      * <li><code>getTargetException()</code>
139      * <li><code>getException()</code>
140      * <li><code>getSourceException()</code>
141      * <li><code>getRootCause()</code>
142      * <li><code>getCausedByException()</code>
143      * <li><code>getNested()</code>
144      * </ul>
145      *
146      * <p>In the absence of any such method, the object is inspected for a
147      * <code>detail</code> field assignable to a <code>Throwable</code>.</p>
148      *
149      * <p>If none of the above is found, returns <code>null</code>.</p>
150      *
151      * @param throwable The exception to introspect for a cause.
152      * @return The cause of the <code>Throwable</code>.
153      * @throws NullPointerException if the throwable is null
154      */
155     public static Throwable getCause( Throwable throwable )
156     {
157         return getCause( throwable, CAUSE_METHOD_NAMES );
158     }

160     /**
161      * <p>Introspects the specified <code>Throwable</code> to obtain the cause
162      * using a supplied array of method names.</p>
163      *
164      * @param throwable The exception to introspect for a cause.
165      * @return The cause of the <code>Throwable</code>.
166      * @throws NullPointerException if the method names array is null or contains null
167      * @throws NullPointerException if the throwable is null
168      */
169     public static Throwable getCause( Throwable throwable, String[] methodNames )
170     {
171         Throwable cause = getCauseUsingWellKnownTypes( throwable );
172         if ( cause == null )
173         {
174             for ( String methodName : methodNames )
175             {
176                 cause = getCauseUsingMethodName( throwable, methodName );
177                 if ( cause != null )
178                 {
179                     break;
180                 }
181             }
182 
183             if ( cause == null )
184             {
185                 cause = getCauseUsingFieldName( throwable, "detail" );
186             }
187         }
188         return cause;
189     }

191     /**
192      * <p>Walks through the exception chain to the last element -- the
193      * "root" of the tree -- using {@link #getCause(Throwable)}, and
194      * returns that exception.</p>
195      *
196      * @param throwable  the throwable to get the root cause for
197      * @return The root cause of the <code>Throwable</code>.
198      */
199     public static Throwable getRootCause( Throwable throwable )
200     {
201         Throwable cause = getCause( throwable );
202         if ( cause != null )
203         {
204             throwable = cause;
205             while ( ( throwable = getCause( throwable ) ) != null )
206             {
207                 cause = throwable;
208             }
209         }
210         return cause;
211     }

213     /**
214      * <p>Uses <code>instanceof</code> checks to examine the exception,
215      * looking for well known types which could contain chained or
216      * wrapped exceptions.</p>
217      *
218      * @param throwable  the exception to examine
219      * @return The wrapped exception, or <code>null</code> if not
220      * found.
221      */
222     private static Throwable getCauseUsingWellKnownTypes( Throwable throwable )
223     {
224         if ( throwable instanceof SQLException )
225         {
226             return ( (SQLException) throwable ).getNextException();
227         }
228         else if ( throwable instanceof InvocationTargetException )
229         {
230             return ( (InvocationTargetException) throwable ).getTargetException();
231         }
232         else
233         {
234             return null;
235         }
236     }

238     /**
239      * <p>Find a throwable by method name.</p>
240      *
241      * @param throwable  the exception to examine
242      * @param methodName  the name of the method to find and invoke
243      * @return The wrapped exception, or <code>null</code> if not
244      * found.
245      */
246     private static Throwable getCauseUsingMethodName( Throwable throwable, String methodName )
247     {
248         Method method = null;
249         try
250         {
251             method = throwable.getClass().getMethod( methodName, null );
252         }
253         catch ( NoSuchMethodException ignored )
254         {
255         }
256         catch ( SecurityException ignored )
257         {
258         }
259 
260         if ( method != null && Throwable.class.isAssignableFrom( method.getReturnType() ) )
261         {
262             try
263             {
264                 return (Throwable) method.invoke( throwable, new Object[0] );
265             }
266             catch ( IllegalAccessException ignored )
267             {
268             }
269             catch ( IllegalArgumentException ignored )
270             {
271             }
272             catch ( InvocationTargetException ignored )
273             {
274             }
275         }
276         return null;
277     }

279     /**
280      * <p>Find a throwable by field name.</p>
281      *
282      * @param throwable  the exception to examine
283      * @param fieldName  the name of the attribute to examine
284      * @return The wrapped exception, or <code>null</code> if not
285      * found.
286      */
287     private static Throwable getCauseUsingFieldName( Throwable throwable, String fieldName )
288     {
289         Field field = null;
290         try
291         {
292             field = throwable.getClass().getField( fieldName );
293         }
294         catch ( NoSuchFieldException ignored )
295         {
296         }
297         catch ( SecurityException ignored )
298         {
299         }
300 
301         if ( field != null && Throwable.class.isAssignableFrom( field.getType() ) )
302         {
303             try
304             {
305                 return (Throwable) field.get( throwable );
306             }
307             catch ( IllegalAccessException ignored )
308             {
309             }
310             catch ( IllegalArgumentException ignored )
311             {
312             }
313         }
314         return null;
315     }

318      * <p>Returns the number of <code>Throwable</code> objects in the
319      * exception chain.</p>
320      *
321      * @param throwable  the exception to inspect
322      * @return The throwable count.
323      */
324     public static int getThrowableCount( Throwable throwable )
325     {
326         // Count the number of throwables
327         int count = 0;
328         while ( throwable != null )
329         {
330             count++;
331             throwable = ExceptionUtils.getCause( throwable );
332         }
333         return count;
334     }

337      * <p>Returns the list of <code>Throwable</code> objects in the
338      * exception chain.</p>
339      *
340      * @param throwable  the exception to inspect
341      * @return The list of <code>Throwable</code> objects.
342      */
343     public static Throwable[] getThrowables( Throwable throwable )
344     {
345         List<Throwable> list = new ArrayList<Throwable>();
346         while ( throwable != null )
347         {
348             list.add( throwable );
349             throwable = getCause( throwable );
350         }
351         return list.toArray( new Throwable[list.size()] );
352     }

365     /**
366      * <p>Returns the (zero based) index, of the first
367      * <code>Throwable</code> that matches the specified type in the
368      * exception chain of <code>Throwable</code> objects with an index
369      * greater than or equal to the specified index, or
370      * <code>-1</code> if the type is not found.</p>
371      *
372      * @param throwable  the exception to inspect
373      * @param type  <code>Class</code> to look for
374      * @param fromIndex  the (zero based) index of the starting
375      *  position in the chain to be searched
376      * @return the first occurrence of the type in the chain, or
377      *  <code>-1</code> if the type is not found
378      * @throws IndexOutOfBoundsException If the <code>fromIndex</code>
379      *  argument is negative or not less than the count of
380      *  <code>Throwable</code>s in the chain.
381      */
382     public static int indexOfThrowable( Throwable throwable, Class type, int fromIndex )
383     {
384         if ( fromIndex < 0 )
385         {
386             throw new IndexOutOfBoundsException( "Throwable index out of range: " + fromIndex );
387         }
388         Throwable[] throwables = ExceptionUtils.getThrowables( throwable );
389         if ( fromIndex >= throwables.length )
390         {
391             throw new IndexOutOfBoundsException( "Throwable index out of range: " + fromIndex );
392         }
393         for ( int i = fromIndex; i < throwables.length; i++ )
394         {
395             if ( throwables[i].getClass().equals( type ) )
396             {
397                 return i;
398             }
399         }
400         return -1;
401     }

403     /**
404      * Prints a compact stack trace for the root cause of a throwable.
405      * The compact stack trace starts with the root cause and prints
406      * stack frames up to the place where it was caught and wrapped.
407      * Then it prints the wrapped exception and continues with stack frames
408      * until the wrapper exception is caught and wrapped again, etc.
409      * <p>
410      * The method is equivalent to t.printStackTrace() for throwables
411      * that don't have nested causes.
412      */
413     public static void printRootCauseStackTrace( Throwable t, PrintStream stream )
414     {
415         String trace[] = getRootCauseStackTrace( t );
416         for ( String aTrace : trace )
417         {
418             stream.println( aTrace );
419         }
420         stream.flush();
421     }

445     /**
446      * Creates a compact stack trace for the root cause of the supplied
447      * throwable.
448      *
449      * See <code>printRootCauseStackTrace(Throwable t, PrintStream s)</code>
450      */
451     public static String[] getRootCauseStackTrace( Throwable t )
452     {
453         Throwable[] throwables = getThrowables( t );
454         int count = throwables.length;
455         ArrayList<String> frames = new ArrayList<String>();
456         List<String> nextTrace = getStackFrameList( throwables[count - 1] );
457         for ( int i = count; --i >= 0; )
458         {
459             List<String> trace = nextTrace;
460             if ( i != 0 )
461             {
462                 nextTrace = getStackFrameList( throwables[i - 1] );
463                 removeCommonFrames( trace, nextTrace );
464             }
465             if ( i == ( count - 1 ) )
466             {
467                 frames.add( throwables[i].toString() );
468             }
469             else
470             {
471                 frames.add( WRAPPED_MARKER + throwables[i].toString() );
472             }
473             for ( String aTrace : trace )
474             {
475                 frames.add( aTrace );
476             }
477         }
478         return frames.toArray( new String[frames.size()] );
479     }

481     /**
482      * Given two stack traces, removes common frames from the cause trace.
483      *
484      * @param causeFrames   stack trace of a cause throwable
485      * @param wrapperFrames stack trace of a wrapper throwable
486      */
487     private static void removeCommonFrames( List<String> causeFrames, List<String> wrapperFrames )
488     {
489         int causeFrameIndex = causeFrames.size() - 1;
490         int wrapperFrameIndex = wrapperFrames.size() - 1;
491         while ( causeFrameIndex >= 0 && wrapperFrameIndex >= 0 )
492         {
493             // Remove the frame from the cause trace if it is the same
494             // as in the wrapper trace
495             String causeFrame = causeFrames.get( causeFrameIndex );
496             String wrapperFrame = wrapperFrames.get( wrapperFrameIndex );
497             if ( causeFrame.equals( wrapperFrame ) )
498             {
499                 causeFrames.remove( causeFrameIndex );
500             }
501             causeFrameIndex--;
502             wrapperFrameIndex--;
503         }
504     }

522     /**
523      * A way to get the entire nested stack-trace of an throwable.
524      *
525      * @param t The <code>Throwable</code>.
526      * @return The nested stack trace, with the root cause first.
527      */
528     public static String getFullStackTrace( Throwable t )
529     {
530         StringWriter sw = new StringWriter();
531         PrintWriter pw = new PrintWriter( sw, true );
532         Throwable[] ts = getThrowables( t );
533         for ( Throwable t1 : ts )
534         {
535             t1.printStackTrace( pw );
536             if ( isNestedThrowable( t1 ) )
537             {
538                 break;
539             }
540         }
541         return sw.getBuffer().toString();
542     }

544     /**
545      * Whether an Throwable is considered nested or not.
546      *
547      * @param throwable The <code>Throwable</code>.
548      * @return boolean true/false
549      */
550     public static boolean isNestedThrowable( Throwable throwable )
551     {
552         if ( throwable == null )
553         {
554             return false;
555         }
556 
557         if ( throwable instanceof SQLException )
558         {
559             return true;
560         }
561         else if ( throwable instanceof InvocationTargetException )
562         {
563             return true;
564         }
565 
566         for ( String CAUSE_METHOD_NAME : CAUSE_METHOD_NAMES )
567         {
568             try
569             {
570                 Method method = throwable.getClass().getMethod( CAUSE_METHOD_NAME, null );
571                 if ( method != null )
572                 {
573                     return true;
574                 }
575             }
576             catch ( NoSuchMethodException ignored )
577             {
578             }
579             catch ( SecurityException ignored )
580             {
581             }
582         }
583 
584         try
585         {
586             Field field = throwable.getClass().getField( "detail" );
587             if ( field != null )
588             {
589                 return true;
590             }
591         }
592         catch ( NoSuchFieldException ignored )
593         {
594         }
595         catch ( SecurityException ignored )
596         {
597         }
598 
599         return false;
600     }

615     /**
616      * Functionality shared between the
617      * <code>getStackFrames(Throwable)</code> methods of this and the
618      * classes.
619      */
620     static String[] getStackFrames( String stackTrace )
621     {
622         String linebreak = System.getProperty( "line.separator" );
623         StringTokenizer frames = new StringTokenizer( stackTrace, linebreak );
624         List<String> list = new LinkedList<String>();
625         while ( frames.hasMoreTokens() )
626         {
627             list.add( frames.nextToken() );
628         }
629         return list.toArray( new String[list.size()]);
630     }

632     /**
633      * Produces a List of stack frames - the message is not included.
634      * This works in most cases - it will only fail if the exception message
635      * contains a line that starts with:  "   at".
636      *
637      * @param t is any throwable
638      * @return List of stack frames
639      */
640     static List<String> getStackFrameList( Throwable t )
641     {
642         String stackTrace = getStackTrace( t );
643         String linebreak = System.getProperty( "line.separator" );
644         StringTokenizer frames = new StringTokenizer( stackTrace, linebreak );
645         List<String> list = new LinkedList<String>();
646         boolean traceStarted = false;
647         while ( frames.hasMoreTokens() )
648         {
649             String token = frames.nextToken();
650             // Determine if the line starts with <whitespace>at
651             int at = token.indexOf( "at" );
652             if ( at != -1 && token.substring( 0, at ).trim().length() == 0 )
653             {
654                 traceStarted = true;
655                 list.add( token );
656             }
657             else if ( traceStarted )
658             {
659                 break;
660             }
661         }
662         return list;
663     }
