100     /**
101      * Description of the Method
102      */
103     protected void expandFile( final File srcF, final File dir )
104         throws Exception
105     {
106         ZipInputStream zis = null;
107         try
108         {
109             // code from WarExpand
110             zis = new ZipInputStream( new FileInputStream( srcF ) );
111 
112             for ( ZipEntry ze = zis.getNextEntry(); ze != null; ze = zis.getNextEntry() )
113             {
114                 extractFile( srcF, dir, zis, ze.getName(), new Date( ze.getTime() ), ze.isDirectory() );
115             }
116 
117             //log("expand complete", Project.MSG_VERBOSE);
118             zis.close();
119             zis = null;
120         }
121         catch ( IOException ioe )
122         {
123             throw new Exception( "Error while expanding " + srcF.getPath(), ioe );
124         }
125         finally
126         {
127             IOUtil.close( zis );
128         }
129     }
130 
131     /**
132      * Description of the Method
133      */
134     protected void extractFile( File srcF, File dir, InputStream compressedInputStream, String entryName,
135                                 Date entryDate, boolean isDirectory )
136         throws Exception
137     {
138         File f = FileUtils.resolveFile( dir, entryName );
139 
140         if ( !f.getAbsolutePath().startsWith( dir.getAbsolutePath() ) )
141         {
142             throw new IOException( "Entry '" + entryName + "' outside the target directory." );
143         }
144 
145         try
146         {
147             if ( !overwrite && f.exists() && f.lastModified() >= entryDate.getTime() )
148             {
149                 return;
150             }
151 
152             // create intermediary directories - sometimes zip don't add them
153             File dirF = f.getParentFile();
154             dirF.mkdirs();
155 
156             if ( isDirectory )
157             {
158                 f.mkdirs();
159             }
160             else
161             {
162                 byte[] buffer = new byte[ 65536 ];
163                 FileOutputStream fos = null;
164                 try
165                 {
166                     fos = new FileOutputStream( f );
167 
168                     for ( int length = compressedInputStream.read( buffer );
169                           length >= 0;
170                           fos.write( buffer, 0, length ), length = compressedInputStream.read( buffer ) );
171 
172                     fos.close();
173                     fos = null;
174                 }
175                 finally
176                 {
177                     IOUtil.close( fos );
178                 }
179             }
180 
181             f.setLastModified( entryDate.getTime() );
182         }
183         catch ( FileNotFoundException ex )
184         {
185             throw new Exception( "Can't extract file " + srcF.getPath(), ex );
186         }
187 
188     }
