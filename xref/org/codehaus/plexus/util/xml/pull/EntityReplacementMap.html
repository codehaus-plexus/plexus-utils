18  public class EntityReplacementMap
19  {
20      final String entityName[];
21  
22      final char[] entityNameBuf[];
23  
24      final String entityReplacement[];
25  
26      final char[] entityReplacementBuf[];
27  
28      int entityEnd;
29  
30      final int entityNameHash[];
31  
32      public EntityReplacementMap( String[][] replacements )
33      {
34          int length = replacements.length;
35          entityName = new String[length];
36          entityNameBuf = new char[length][];
37          entityReplacement = new String[length];
38          entityReplacementBuf = new char[length][];
39          entityNameHash = new int[length];
40  
41          for ( String[] replacement : replacements )
42          {
43              defineEntityReplacementText( replacement[0], replacement[1] );
44          }
45      }
46  
47      private void defineEntityReplacementText( String entityName, String replacementText )
48      {
49          if ( !replacementText.startsWith( "&#" ) && this.entityName != null && replacementText.length() > 1 )
50          {
51              String tmp = replacementText.substring( 1, replacementText.length() - 1 );
52              for ( int i = 0; i < this.entityName.length; i++ )
53              {
54                  if ( this.entityName[i] != null && this.entityName[i].equals( tmp ) )
55                  {
56                      replacementText = this.entityReplacement[i];
57                  }
58              }
59          }
60  
61          // this is to make sure that if interning works we will take advantage of it ...
62          char[] entityNameCharData = entityName.toCharArray();
63          // noinspection ConstantConditions
64          this.entityName[entityEnd] = newString( entityNameCharData, 0, entityName.length() );
65          entityNameBuf[entityEnd] = entityNameCharData;
66  
67          entityReplacement[entityEnd] = replacementText;
68          entityReplacementBuf[entityEnd] = replacementText.toCharArray();
69          entityNameHash[entityEnd] = fastHash( entityNameBuf[entityEnd], 0, entityNameBuf[entityEnd].length );
70          ++entityEnd;
71          // TODO disallow < or & in entity replacement text (or ]]>???)
72          // TODO keepEntityNormalizedForAttributeValue cached as well ...
73      }
74  
75      private String newString( char[] cbuf, int off, int len )
76      {
77          return new String( cbuf, off, len );
78      }
79  
80      /**
81       * simplistic implementation of hash function that has <b>constant</b> time to compute - so it also means
82       * diminishing hash quality for long strings but for XML parsing it should be good enough ...
83       */
84      private static int fastHash( char ch[], int off, int len )
85      {
86          if ( len == 0 )
87              return 0;
88          // assert len >0
89          int hash = ch[off]; // hash at beginning
90          // try {
91          hash = ( hash << 7 ) + ch[off + len - 1]; // hash at the end
92          // } catch(ArrayIndexOutOfBoundsException aie) {
93          // aie.printStackTrace(); //should never happen ...
94          // throw new RuntimeException("this is violation of pre-condition");
95          // }
96          if ( len > 16 )
97              hash = ( hash << 7 ) + ch[off + ( len / 4 )]; // 1/4 from beginning
98          if ( len > 8 )
99              hash = ( hash << 7 ) + ch[off + ( len / 2 )]; // 1/2 of string size ...
100         // notice that hash is at most done 3 times <<7 so shifted by 21 bits 8 bit value
101         // so max result == 29 bits so it is quite just below 31 bits for long (2^32) ...
102         // assert hash >= 0;
103         return hash;
104     }
