148     protected void ensureElementsCapacity() {
149         final int elStackSize = elName.length;
150         //assert (depth + 1) >= elName.length;
151         // we add at least one extra slot ...
152         final int newSize = (depth >= 7 ? 2 * depth : 8) + 2; // = lucky 7 + 1 //25
153         if(TRACE_SIZING) {
154             System.err.println(
155                 getClass().getName()+" elStackSize "+elStackSize+" ==> "+newSize);
156         }
157         final boolean needsCopying = elStackSize > 0;
158         String[] arr = null;
159         // reuse arr local variable slot
160         arr = new String[newSize];
161         if(needsCopying) System.arraycopy(elName, 0, arr, 0, elStackSize);
162         elName = arr;
163         arr = new String[newSize];
164         if(needsCopying) System.arraycopy(elNamespace, 0, arr, 0, elStackSize);
165         elNamespace = arr;
166 
167         final int[] iarr = new int[newSize];
168         if(needsCopying) {
169             System.arraycopy(elNamespaceCount, 0, iarr, 0, elStackSize);
170         } else {
171             // special initialization
172             iarr[0] = 0;
173         }
174         elNamespaceCount = iarr;
175     }

215                            boolean state) throws IllegalArgumentException, IllegalStateException
216     {
217         if(name == null) {
218             throw new IllegalArgumentException("feature name can not be null");
219         }
220         if(FEATURE_NAMES_INTERNED.equals(name)) {

222         } else if(FEATURE_SERIALIZER_ATTVALUE_USE_APOSTROPHE.equals(name)) {

225             throw new IllegalStateException("unsupported feature "+name);

251     /**
252      * For maximum efficiency when writing indents the required output is pre-computed
253      * This is internal function that recomputes buffer after user requested chnages.
254      */
255     protected void rebuildIndentationBuf() {
256         if(doIndent == false) return;
257         final int maxIndent = 65; //hardcoded maximum indentation size in characters
258         int bufSize = 0;
259         offsetNewLine = 0;
260         if(writeLineSepartor) {
261             offsetNewLine = lineSeparator.length();
262             bufSize += offsetNewLine;
263         }
264         maxIndentLevel = 0;
265         if(writeIndentation) {
266             indentationJump = indentationString.length();
267             maxIndentLevel = maxIndent / indentationJump;
268             bufSize += maxIndentLevel * indentationJump;
269         }
270         if(indentationBuf == null || indentationBuf.length < bufSize) {
271             indentationBuf = new char[bufSize + 8];
272         }
273         int bufPos = 0;
274         if(writeLineSepartor) {
275             for (int i = 0; i < lineSeparator.length(); i++)
276             {
277                 indentationBuf[ bufPos++ ] = lineSeparator.charAt(i);
278             }
279         }
280         if(writeIndentation) {
281             for (int i = 0; i < maxIndentLevel; i++)
282             {
283                 for (int j = 0; j < indentationString.length(); j++)
284                 {
285                     indentationBuf[ bufPos++ ] = indentationString.charAt(j);
286                 }
287             }
288         }
289     }

291     // if(doIndent) writeIndent();
292     protected void writeIndent() throws IOException {
293         final int start = writeLineSepartor ? 0 : offsetNewLine;
294         final int level = (depth > maxIndentLevel) ? maxIndentLevel : depth;
295         out.write( indentationBuf, start, (level * indentationJump) + offsetNewLine);
296     }

298     public void setProperty(String name,
299                             Object value) throws IllegalArgumentException, IllegalStateException
300     {
301         if(name == null) {
302             throw new IllegalArgumentException("property name can not be null");
303         }
304         if(PROPERTY_SERIALIZER_INDENTATION.equals(name)) {
305             indentationString = (String)value;
306         } else if(PROPERTY_SERIALIZER_LINE_SEPARATOR.equals(name)) {
307             lineSeparator = (String)value;
308         } else if(PROPERTY_LOCATION.equals(name)) {
309             location = (String) value;
310         } else {
311             throw new IllegalStateException("unsupported property "+name);
312         }
313         writeLineSepartor = lineSeparator != null && lineSeparator.length() > 0;
314         writeIndentation = indentationString != null && indentationString.length() > 0;
315         // optimize - do not write when nothing to write ...
316         doIndent = indentationString != null && (writeLineSepartor || writeIndentation);
317         //NOTE: when indentationString == null there is no indentation
318         //      (even though writeLineSeparator may be true ...)
319         rebuildIndentationBuf();
320         seenTag = false; // for consistency
321     }

402     public void endDocument() throws IOException
403     {
404         // close all unclosed tag;
405         while(depth > 0) {
406             endTag(elNamespace[ depth ], elName[ depth ]);
407         }
408         if(writeLineSepartor) {
409             out.write(lineSeparator);
410         }
411         //assert depth == 0;
412         //assert startTagIncomplete == false;
413         finished = pastRoot = startTagIncomplete = true;
414         out.flush();
415     }

417     public void setPrefix(String prefix, String namespace) throws IOException
418     {
419         if(startTagIncomplete) closeStartTag();
420         //assert prefix != null;
421         //assert namespace != null;
422         if (prefix == null) {
423             prefix = "";
424         }
425         if(!namesInterned) {
426             prefix = prefix.intern(); //will throw NPE if prefix==null
427         } else if(checkNamesInterned) {
428             checkInterning(prefix);
429         } else if(prefix == null) {
430             throw new IllegalArgumentException("prefix must be not null"+getLocation());
431         }
432 
433         //check that prefix is not duplicated ...
434         for (int i = elNamespaceCount[ depth ]; i < namespaceEnd; i++)
435         {
436             if(prefix == namespacePrefix[ i ]) {
437                 throw new IllegalStateException("duplicated prefix "+printable(prefix)+getLocation());
438             }
439         }
440 
441         if(!namesInterned) {
442             namespace = namespace.intern();
443         } else if(checkNamesInterned) {
444             checkInterning(namespace);
445         } else if(namespace == null) {
446             throw new IllegalArgumentException("namespace must be not null"+getLocation());
447         }
448 
449         if(namespaceEnd >= namespacePrefix.length) {
450             ensureNamespacesCapacity();
451         }
452         namespacePrefix[ namespaceEnd ] = prefix;
453         namespaceUri[ namespaceEnd ] = namespace;
454         ++namespaceEnd;
455         setPrefixCalled = true;
456     }

462     public String getPrefix(String namespace, boolean generatePrefix)
463     {
464         //assert namespace != null;
465         if(!namesInterned) {
466             // when String is interned we can do much faster namespace stack lookups ...
467             namespace = namespace.intern();
468         } else if(checkNamesInterned) {
469             checkInterning(namespace);
470             //assert namespace != namespace.intern();
471         }
472         if(namespace == null) {
473             throw new IllegalArgumentException("namespace must be not null"+getLocation());
474         } else if(namespace.length() == 0) {
475             throw new IllegalArgumentException("default namespace cannot have prefix"+getLocation());
476         }
477 
478         // first check if namespace is already in scope
479         for (int i = namespaceEnd - 1; i >= 0 ; --i)
480         {
481             if(namespace == namespaceUri[ i ]) {
482                 final String prefix = namespacePrefix[ i ];
483                 // now check that prefix is still in scope
484                 for (int p = namespaceEnd - 1; p > i ; --p)
485                 {
486                     if(prefix == namespacePrefix[ p ])
487                         continue; // too bad - prefix is redeclared with different namespace
488                 }
489                 return prefix;
490             }
491         }
492 
493         // so not found it ...
494         if(!generatePrefix) {
495             return null;
496         }
497         return generatePrefix(namespace);
498     }

500     private String generatePrefix(String namespace) {
501         //assert namespace == namespace.intern();
502         while(true) {
503             ++autoDeclaredPrefixes;
504             //fast lookup uses table that was pre-initialized in static{} ....
505             final String prefix = autoDeclaredPrefixes < precomputedPrefixes.length
506                 ? precomputedPrefixes[autoDeclaredPrefixes] : ("n"+autoDeclaredPrefixes).intern();
507             // make sure this prefix is not declared in any scope (avoid hiding in-scope prefixes)!
508             for (int i = namespaceEnd - 1; i >= 0 ; --i)
509             {
510                 if(prefix == namespacePrefix[ i ]) {
511                     continue; // prefix is already declared - generate new and try again
512                 }
513             }
514             // declare prefix
515 
516             if(namespaceEnd >= namespacePrefix.length) {
517                 ensureNamespacesCapacity();
518             }
519             namespacePrefix[ namespaceEnd ] = prefix;
520             namespaceUri[ namespaceEnd ] = namespace;
521             ++namespaceEnd;
522 
523             return prefix;
524         }
525     }

542     public XmlSerializer startTag (String namespace, String name) throws IOException
543     {
544 
545         if(startTagIncomplete) {
546             closeStartTag();
547         }
548         seenBracket = seenBracketBracket = false;
549         if(doIndent && depth > 0 && seenTag) {
550             writeIndent();
551         }
552         seenTag = true;
553         setPrefixCalled = false;
554         startTagIncomplete = true;
555         ++depth;
556         if( (depth + 1) >= elName.length) {
557             ensureElementsCapacity();
558         }
559         ////assert namespace != null;
560 
561         if(checkNamesInterned && namesInterned) checkInterning(namespace);
562         elNamespace[ depth ] = (namesInterned || namespace == null) ? namespace : namespace.intern();
563         //assert name != null;
564         //elName[ depth ] = name;
565         if(checkNamesInterned && namesInterned) checkInterning(name);
566         elName[ depth ] = (namesInterned || name == null) ? name : name.intern();
567         if(out == null) {
568             throw new IllegalStateException("setOutput() must called set before serialization can start");
569         }
570         out.write('<');
571         if(namespace != null)  {
572 
573             if(namespace.length() > 0) {
574                 //ALEK: in future make it as feature on serializer
575                 String prefix = null;
576                 if(depth > 0 && (namespaceEnd - elNamespaceCount[depth-1]) == 1) {
577                     // if only one prefix was declared un-declare it if prefix is already declared on parent el with the same URI
578                     String uri = namespaceUri[namespaceEnd-1];
579                     if(uri == namespace || uri.equals(namespace)) {
580                         String elPfx = namespacePrefix[namespaceEnd-1];
581                         // 2 == to skip predefined namespaces (xml and xmlns ...)
582                         for(int pos = elNamespaceCount[depth-1] - 1; pos >= 2; --pos ) {
583                             String pf = namespacePrefix[pos];
584                             if(pf == elPfx || pf.equals(elPfx)) {
585                                 String n = namespaceUri[pos];
586                                 if(n == uri || n.equals(uri)) {
587                                     --namespaceEnd; //un-declare namespace
588                                     prefix = elPfx;
589                                 }
590                                 break;
591                             }
592                         }
593                     }
594                 }
595                 if(prefix == null) {
596                     prefix = lookupOrDeclarePrefix( namespace );
597                 }
598                 //assert prefix != null;
599                 // make sure that default ("") namespace to not print ":"
600                 if(prefix.length() > 0) {
601                     out.write(prefix);
602                     out.write(':');
603                 }
604             } else {
605                 // make sure that default namespace can be declared
606                 for (int i = namespaceEnd - 1; i >= 0 ; --i)
607                 {
608                     if(namespacePrefix[ i ] == "") {
609                         final String uri = namespaceUri[ i ];
610                         if(uri == null) {
611                             // declare default namespace
612                             setPrefix("", "");
613                         } else if(uri.length() > 0) {
614                             throw new IllegalStateException(
615                                 "start tag can not be written in empty default namespace "+
616                                     "as default namespace is currently bound to '"+uri+"'"+getLocation());
617                         }
618                         break;
619                     }
620                 }
621             }
622 
623         }
624         out.write(name);
625         return this;
626     }

628     public XmlSerializer attribute (String namespace, String name,
629                                     String value) throws IOException
630     {
631         if(!startTagIncomplete) {
632             throw new IllegalArgumentException("startTag() must be called before attribute()"+getLocation());
633         }
634         //assert setPrefixCalled == false;
635         out.write(' ');
636         ////assert namespace != null;
637         if(namespace != null && namespace.length() > 0) {
638             //namespace = namespace.intern();
639             if(!namesInterned) {
640                 namespace = namespace.intern();
641             } else if(checkNamesInterned) {
642                 checkInterning(namespace);
643             }
644             String prefix = lookupOrDeclarePrefix( namespace );
645             //assert( prefix != null);
646             if(prefix.length() == 0) {
647                 // needs to declare prefix to hold default namespace
648                 //NOTE: attributes such as a='b' are in NO namespace
649                 prefix = generatePrefix(namespace);
650             }
651             out.write(prefix);
652             out.write(':');
653             //            if(prefix.length() > 0) {
654             //                out.write(prefix);
655             //                out.write(':');
656             //            }
657         }
658         //assert name != null;
659         out.write(name);
660         out.write('=');
661         //assert value != null;
662         out.write( attributeUseApostrophe ? '\'' : '"');
663         writeAttributeValue(value, out);
664         out.write( attributeUseApostrophe ? '\'' : '"');
665         return this;
666     }

668     protected void closeStartTag() throws IOException {
669         if(finished) {
670             throw new IllegalArgumentException("trying to write past already finished output"+getLocation());
671         }
672         if(seenBracket) {
673             seenBracket = seenBracketBracket = false;
674         }
675         if( startTagIncomplete || setPrefixCalled ) {
676             if(setPrefixCalled) {
677                 throw new IllegalArgumentException(
678                     "startTag() must be called immediately after setPrefix()"+getLocation());
679             }
680             if(!startTagIncomplete) {
681                 throw new IllegalArgumentException("trying to close start tag that is not opened"+getLocation());
682             }
683 
684             // write all namespace declarations!
685             writeNamespaceDeclarations();
686             out.write('>');
687             elNamespaceCount[ depth ] = namespaceEnd;
688             startTagIncomplete = false;
689         }
690     }

692     private void writeNamespaceDeclarations() throws IOException
693     {
694         //int start = elNamespaceCount[ depth - 1 ];
695         for (int i = elNamespaceCount[ depth - 1 ]; i < namespaceEnd; i++)
696         {
697             if(doIndent && namespaceUri[ i ].length() > 40) {
698                 writeIndent();
699                 out.write(" ");
700             }
701             if(namespacePrefix[ i ] != "") {
702                 out.write(" xmlns:");
703                 out.write(namespacePrefix[ i ]);
704                 out.write('=');
705             } else {
706                 out.write(" xmlns=");
707             }
708             out.write( attributeUseApostrophe ? '\'' : '"');
709 
710             //NOTE: escaping of namespace value the same way as attributes!!!!
711             writeAttributeValue(namespaceUri[ i ], out);
712 
713             out.write( attributeUseApostrophe ? '\'' : '"');
714         }

717     public XmlSerializer endTag(String namespace, String name) throws IOException
718     {
719         // check that level is valid
720         ////assert namespace != null;
721         //if(namespace != null) {
722         //    namespace = namespace.intern();
723         //}
724         seenBracket = seenBracketBracket = false;
725         if(namespace != null) {
726             if(!namesInterned) {
727                 namespace = namespace.intern();
728             } else if(checkNamesInterned) {
729                 checkInterning(namespace);
730             }
731         }
732 
733         if(namespace != elNamespace[ depth ])
734         {
735             throw new IllegalArgumentException(
736                 "expected namespace "+printable(elNamespace[ depth ])
737                     +" and not "+printable(namespace)+getLocation());
738         }
739         if(name == null) {
740             throw new IllegalArgumentException("end tag name can not be null"+getLocation());
741         }
742         if(checkNamesInterned && namesInterned) {
743             checkInterning(name);
744         }
745 
746         if((!namesInterned && !name.equals(elName[ depth ]))
747                || (namesInterned && name != elName[ depth ]))
748         {
749             throw new IllegalArgumentException(
750                 "expected element name "+printable(elName[ depth ])+" and not "+printable(name)+getLocation());
751         }
752         if(startTagIncomplete) {
753             writeNamespaceDeclarations();
754             out.write(" />"); //space is added to make it easier to work in XHTML!!!
755             --depth;
756         } else {
757             --depth;
758             //assert startTagIncomplete == false;
759             if(doIndent && seenTag) { writeIndent(); }
760             out.write("</");
761             if(namespace != null && namespace.length() > 0) {
762                 //TODO prefix should be alredy known from matching start tag ...
763                 final String prefix = lookupOrDeclarePrefix( namespace );
764                 //assert( prefix != null);
765                 if(prefix.length() > 0) {
766                     out.write(prefix);
767                     out.write(':');
768                 }
769             }
770             out.write(name);
771             out.write('>');
772 
773         }
774         namespaceEnd = elNamespaceCount[ depth ];
775         startTagIncomplete = false;
776         seenTag = true;
777         return this;
778     }

861     protected void writeAttributeValue(String value, Writer out) throws IOException
862     {
863         //.[apostrophe and <, & escaped],
864         final char quot = attributeUseApostrophe ? '\'' : '"';
865         final String quotEntity = attributeUseApostrophe ? "&apos;" : "&quot;";
866 
867         int pos = 0;
868         for (int i = 0; i < value.length(); i++)
869         {
870             char ch = value.charAt(i);
871             if(ch == '&') {
872                 if(i > pos) out.write(value.substring(pos, i));
873                 out.write("&amp;");
874                 pos = i + 1;
875             } if(ch == '<') {
876                 if(i > pos) out.write(value.substring(pos, i));
877                 out.write("&lt;");
878                 pos = i + 1;
879             }else if(ch == quot) {
880                 if(i > pos) out.write(value.substring(pos, i));
881                 out.write(quotEntity);
882                 pos = i + 1;
883             } else if(ch < 32) {
884                 //in XML 1.0 only legal character are #x9 | #xA | #xD
885                 // and they must be escaped otherwise in attribute value they are normalized to spaces
886                 if(ch == 13 || ch == 10 || ch == 9) {
887                     if(i > pos) out.write(value.substring(pos, i));
888                     out.write("&#");
889                     out.write(Integer.toString(ch));
890                     out.write(';');
891                     pos = i + 1;
892                 } else {
893                     throw new IllegalStateException(
894                         "character "+Integer.toString(ch)+" is not allowed in output"+getLocation());
895                     // in XML 1.1 legal are [#x1-#xD7FF]
896                     //              if(ch > 0) {
897                     //                  if(i > pos) out.write(text.substring(pos, i));
898                     //                  out.write("&#");
899                     //                  out.write(Integer.toString(ch));
900                     //                  out.write(';');
901                     //                  pos = i + 1;
902                     //              } else {
903                     //              throw new IllegalStateException(
904                     //                  "character zero is not allowed in XML 1.1 output"+getLocation());
905                     //              }
906                 }
907             }
908         }
909         if(pos > 0) {
910             out.write(value.substring(pos));
911         } else {
912             out.write(value);  // this is shortcut to the most common case
913         }

1081     private static void addPrintable(StringBuilder retval, char ch)
1082     {
1083         switch (ch)
1084         {
1085             case '\b':
1086                 retval.append("\\b");
1087                 break;
1088             case '\t':
1089                 retval.append("\\t");
1090                 break;
1091             case '\n':
1092                 retval.append("\\n");
1093                 break;
1094             case '\f':
1095                 retval.append("\\f");
1096                 break;
1097             case '\r':
1098                 retval.append("\\r");
1099                 break;
1100             case '\"':
1101                 retval.append("\\\"");
1102                 break;
1103             case '\'':
1104                 retval.append("\\\'");
1105                 break;
1106             case '\\':
1107                 retval.append("\\\\");
1108                 break;
1109             default:
1110                 if (ch  < 0x20 || ch > 0x7e) {
1111                     final String ss = "0000" + Integer.toString(ch, 16);
1112                     retval.append( "\\u" ).append( ss, ss.length() - 4, ss.length());
1113                 } else {
1114                     retval.append(ch);
1115                 }
1116         }
1117     }
