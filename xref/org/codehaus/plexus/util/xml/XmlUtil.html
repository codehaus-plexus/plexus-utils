50      /**
51       * Determines if a given File shall be handled as XML.
52       *
53       * @param f not null file
54       * @return <code>true</code> if the given file has XML content, <code>false</code> otherwise.
55       */
56      public static boolean isXml( File f )
57      {
58          if ( f == null )
59          {
60              throw new IllegalArgumentException( "f could not be null." );
61          }
62  
63          if ( !f.isFile() )
64          {
65              throw new IllegalArgumentException( "The file '" + f.getAbsolutePath() + "' is not a file." );
66          }
67  
68          Reader reader = null;
69          try
70          {
71              reader = ReaderFactory.newXmlReader( f );
72              XmlPullParser parser = new MXParser();
73              parser.setInput( reader );
74              parser.nextToken();
75  
76              return true;
77          }
78          catch ( Exception e )
79          {
80              return false;
81          }
82          finally
83          {
84              IOUtil.close( reader );
85          }
86      }

115      * <pre>

117      * </pre>

123      * </pre>
124      *
125      * @param reader not null
126      * @param writer not null
127      * @param indentSize positive number for the indentation
128      * @param lineSeparator the wanted line separator
129      * @throws IOException if any or invalid xml content
130      * @see ReaderFactory to read an xml content
131      * @see WriterFactory to write an xml content
132      */
133     public static void prettyFormat( Reader reader, Writer writer, int indentSize, String lineSeparator )
134         throws IOException
135     {
136         if ( reader == null )
137         {
138             throw new IllegalArgumentException( "The reader is null" );
139         }
140         if ( writer == null )
141         {
142             throw new IllegalArgumentException( "The writer is null" );
143         }
144         if ( indentSize < 0 )
145         {
146             indentSize = 0;
147         }
148 
149         PrettyPrintXMLWriter xmlWriter = new PrettyPrintXMLWriter( writer );
150         xmlWriter.setLineIndenter( StringUtils.repeat( " ", indentSize ) );
151         xmlWriter.setLineSeparator( lineSeparator );
152 
153         XmlPullParser parser = new MXParser();
154         try
155         {
156             parser.setInput( reader );
157 
158             prettyFormatInternal( parser, xmlWriter );
159         }
160         catch ( XmlPullParserException e )
161         {
162             throw new IOException( "Unable to parse the XML: " + e.getMessage() );
163         }

191      * <pre>
192      * &lt;div&gt;&lt;b&gt;content&lt;/b&gt;&lt;/div&gt;
193      * </pre>
194      * becomes
195      * <pre>
196      * &lt;div&gt;
197      *   &lt;b&gt;content&lt;/b&gt;
198      * &lt;/div&gt;
199      * </pre>
200      *
201      * @param is not null
202      * @param os not null
203      * @param indentSize positive number for the indentation
204      * @param lineSeparator the wanted line separator
205      * @throws IOException if any or invalid xml content
206      */
207     public static void prettyFormat( InputStream is, OutputStream os, int indentSize, String lineSeparator )
208         throws IOException
209     {
210         if ( is == null )
211         {
212             throw new IllegalArgumentException( "The is is null" );
213         }
214         if ( os == null )
215         {
216             throw new IllegalArgumentException( "The os is null" );
217         }
218         if ( indentSize < 0 )
219         {
220             indentSize = 0;
221         }
222 
223         Reader reader = null;
224 
225         Writer out = new OutputStreamWriter( os );
226         PrettyPrintXMLWriter xmlWriter = new PrettyPrintXMLWriter( out );
227         xmlWriter.setLineIndenter( StringUtils.repeat( " ", indentSize ) );
228         xmlWriter.setLineSeparator( lineSeparator );
229 
230         XmlPullParser parser = new MXParser();
231         try
232         {
233             reader = ReaderFactory.newXmlReader( is );
234 
235             parser.setInput( reader );
236 
237             prettyFormatInternal( parser, xmlWriter );
238         }
239         catch ( XmlPullParserException e )
240         {
241             throw new IOException( "Unable to parse the XML: " + e.getMessage() );
242         }
243         finally
244         {
245             IOUtil.close( reader );
246             IOUtil.close( out );
247         }
248     }

250     /**
251      * @param parser not null
252      * @param writer not null
253      * @throws XmlPullParserException if any
254      * @throws IOException if any
255      */
256     private static void prettyFormatInternal( XmlPullParser parser, PrettyPrintXMLWriter writer )
257         throws XmlPullParserException, IOException
258     {
259         boolean hasTag = false;
260         boolean hasComment = false;
261         int eventType = parser.getEventType();
262         while ( eventType != XmlPullParser.END_DOCUMENT )
263         {
264             if ( eventType == XmlPullParser.START_TAG )
265             {
266                 hasTag = true;
267                 if ( hasComment )
268                 {
269                     writer.writeText( writer.getLineIndenter() );
270                     hasComment = false;
271                 }
272                 writer.startElement( parser.getName() );
273                 for ( int i = 0; i < parser.getAttributeCount(); i++ )
274                 {
275                     String key = parser.getAttributeName( i );
276                     String value = parser.getAttributeValue( i );
277                     writer.addAttribute( key, value );
278                 }
279             }
280             else if ( eventType == XmlPullParser.TEXT )
281             {
282                 String text = parser.getText();
283                 if ( !text.trim().equals( "" ) )
284                 {
285                     text = StringUtils.removeDuplicateWhitespace( text );
286                     writer.writeText( text );
287                 }
288             }
289             else if ( eventType == XmlPullParser.END_TAG )
290             {
291                 hasTag = false;
292                 writer.endElement();
293             }
294             else if ( eventType == XmlPullParser.COMMENT )
295             {
296                 hasComment = true;
297                 if ( !hasTag )
298                 {
299                     writer.writeMarkup( writer.getLineSeparator() );
300                     for ( int i = 0; i < writer.getDepth(); i++ )
301                     {
302                         writer.writeMarkup( writer.getLineIndenter() );
303                     }
304                 }
305                 writer.writeMarkup( "<!--" + parser.getText().trim() + " -->" );
306                 if ( !hasTag )
307                 {
308                     writer.writeMarkup( writer.getLineSeparator() );
309                     for ( int i = 0; i < writer.getDepth() - 1; i++ )
310                     {
311                         writer.writeMarkup( writer.getLineIndenter() );
312                     }
313                 }
314             }
315             else if ( eventType == XmlPullParser.DOCDECL )
316             {
317                 writer.writeMarkup( "<!DOCTYPE" + parser.getText() + ">" );
318                 writer.endOfLine();
319             }
320             else if ( eventType == XmlPullParser.PROCESSING_INSTRUCTION )
321             {
322                 writer.writeMarkup( "<?" + parser.getText() + "?>" );
323                 writer.endOfLine();
324             }
325             else if ( eventType == XmlPullParser.CDSECT )
326             {
327                 writer.writeMarkup( "<![CDATA[" + parser.getText() + "]]>" );
328             }
329             else if ( eventType == XmlPullParser.ENTITY_REF )
330             {
331                 writer.writeMarkup( "&" + parser.getName() + ";" );
332             }
333 
334             eventType = parser.nextToken();
335         }
336     }
