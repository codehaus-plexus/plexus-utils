66      /**
67       * This default mode for combining children DOMs during merge means that where element names match, the process will
68       * try to merge the element data, rather than putting the dominant and recessive elements (which share the same
69       * element name) as siblings in the resulting DOM.
70       */
71      public static final String DEFAULT_CHILDREN_COMBINATION_MODE = CHILDREN_COMBINATION_MERGE;

81      /**
82       * This default mode for combining a DOM node during merge means that where element names match, the process will
83       * try to merge the element attributes and values, rather than overriding the recessive element completely with the
84       * dominant one. This means that wherever the dominant element doesn't provide the value or a particular attribute,
85       * that value or attribute will be set from the recessive DOM node.
86       */
87      public static final String DEFAULT_SELF_COMBINATION_MODE = SELF_COMBINATION_MERGE;

115     /**
116      * Copy constructor with alternative name.
117      * @param src The source Dom.
118      * @param name The name of the Dom.
119      */
120     public Xpp3Dom( Xpp3Dom src, String name )
121     {
122         this.name = name;
123         this.inputLocation = src.inputLocation;
124 
125         int childCount = src.getChildCount();
126 
127         childList = new ArrayList<Xpp3Dom>( childCount );
128 
129         setValue( src.getValue() );
130 
131         String[] attributeNames = src.getAttributeNames();
132         for ( String attributeName : attributeNames )
133         {
134             setAttribute( attributeName, src.getAttribute( attributeName ) );
135         }
136 
137         for ( int i = 0; i < childCount; i++ )
138         {
139             addChild( new Xpp3Dom( src.getChild( i ) ) );
140         }
141     }

231     public Xpp3Dom getChild( String name )
232     {
233         if ( name != null )
234         {
235             ListIterator<Xpp3Dom> it = childList.listIterator( childList.size() );
236             while ( it.hasPrevious() )
237             {
238                 Xpp3Dom child = it.previous();
239                 if ( name.equals( child.getName() ) )
240                 {
241                     return child;
242                 }
243             }
244         }
245         return null;
246     }

271     private List<Xpp3Dom> getChildrenAsList( String name )
272     {
273         if ( null == childList )
274         {
275             return Collections.emptyList();
276         }
277         else
278         {
279             ArrayList<Xpp3Dom> children = null;
280 
281             for ( Xpp3Dom configuration : childList )
282             {
283                 if ( name.equals( configuration.getName() ) )
284                 {
285                     if ( children == null )
286                     {
287                         children = new ArrayList<Xpp3Dom>();
288                     }
289                     children.add( configuration );
290                 }
291             }
292 
293             if ( children != null )
294             {
295                 return children;
296             }
297             else
298             {
299                 return Collections.emptyList();
300             }
301         }
302     }

382     /**
383      * Merges one DOM into another, given a specific algorithm and possible override points for that algorithm.<p>
384      * The algorithm is as follows:
385      * <ol>
386      * <li> if the recessive DOM is null, there is nothing to do... return.</li>
387      * <li> Determine whether the dominant node will suppress the recessive one (flag=mergeSelf).
388      *   <ol type="A">
389      *   <li> retrieve the 'combine.self' attribute on the dominant node, and try to match against 'override'...
390      *        if it matches 'override', then set mergeSelf == false...the dominant node suppresses the recessive one
391      *        completely.</li>
392      *   <li> otherwise, use the default value for mergeSelf, which is true...this is the same as specifying
393      *        'combine.self' == 'merge' as an attribute of the dominant root node.</li>
394      *   </ol></li>
395      * <li> If mergeSelf == true
396      *   <ol type="A">
397      *   <li> if the dominant root node's value is empty, set it to the recessive root node's value</li>
398      *   <li> For each attribute in the recessive root node which is not set in the dominant root node, set it.</li>
399      *   <li> Determine whether children from the recessive DOM will be merged or appended to the dominant DOM as
400      *        siblings (flag=mergeChildren).
401      *     <ol type="i">
402      *     <li> if childMergeOverride is set (non-null), use that value (true/false)</li>
403      *     <li> retrieve the 'combine.children' attribute on the dominant node, and try to match against
404      *          'append'...</li>
405      *     <li> if it matches 'append', then set mergeChildren == false...the recessive children will be appended as
406      *          siblings of the dominant children.</li>
407      *     <li> otherwise, use the default value for mergeChildren, which is true...this is the same as specifying
408      *         'combine.children' == 'merge' as an attribute on the dominant root node.</li>
409      *     </ol></li>
410      *   <li> Iterate through the recessive children, and:
411      *     <ol type="i">
412      *     <li> if mergeChildren == true and there is a corresponding dominant child (matched by element name),
413      *          merge the two.</li>
414      *     <li> otherwise, add the recessive child as a new child on the dominant root node.</li>
415      *     </ol></li>
416      *   </ol></li>
417      * </ol>
418      */
419     private static void mergeIntoXpp3Dom( Xpp3Dom dominant, Xpp3Dom recessive, Boolean childMergeOverride )
420     {
421         // TODO: share this as some sort of assembler, implement a walk interface?
422         if ( recessive == null )
423         {
424             return;
425         }
426 
427         boolean mergeSelf = true;
428 
429         String selfMergeMode = dominant.getAttribute( SELF_COMBINATION_MODE_ATTRIBUTE );
430 
431         if ( SELF_COMBINATION_OVERRIDE.equals( selfMergeMode ) )
432         {
433             mergeSelf = false;
434         }
435 
436         if ( mergeSelf )
437         {
438             if ( isEmpty( dominant.getValue() ) && !isEmpty( recessive.getValue() ) )
439             {
440                 dominant.setValue( recessive.getValue() );
441                 dominant.setInputLocation( recessive.getInputLocation() );
442             }
443 
444             if ( recessive.attributes != null )
445             {
446                 for ( String attr : recessive.attributes.keySet() )
447                 {
448                     if ( isEmpty( dominant.getAttribute( attr ) ) && !SELF_COMBINATION_MODE_ATTRIBUTE.equals( attr ) )
449                     {
450                         dominant.setAttribute( attr, recessive.getAttribute( attr ) );
451                     }
452                 }
453             }
454 
455             if ( recessive.getChildCount() > 0 )
456             {
457                 boolean mergeChildren = true;
458 
459                 if ( childMergeOverride != null )
460                 {
461                     mergeChildren = childMergeOverride;
462                 }
463                 else
464                 {
465                     String childMergeMode = dominant.getAttribute( CHILDREN_COMBINATION_MODE_ATTRIBUTE );
466 
467                     if ( CHILDREN_COMBINATION_APPEND.equals( childMergeMode ) )
468                     {
469                         mergeChildren = false;
470                     }
471                 }
472 
473                 if ( !mergeChildren )
474                 {
475                     Xpp3Dom[] dominantChildren = dominant.getChildren();
476                     // remove these now, so we can append them to the recessive list later.
477                     dominant.childList.clear();
478 
479                     for ( int i = 0, recessiveChildCount = recessive.getChildCount(); i < recessiveChildCount; i++ )
480                     {
481                         Xpp3Dom recessiveChild = recessive.getChild( i );
482                         dominant.addChild( new Xpp3Dom( recessiveChild ) );
483                     }
484 
485                     // now, re-add these children so they'll be appended to the recessive list.
486                     for ( Xpp3Dom aDominantChildren : dominantChildren )
487                     {
488                         dominant.addChild( aDominantChildren );
489                     }
490                 }
491                 else
492                 {
493                     Map<String, Iterator<Xpp3Dom>> commonChildren = new HashMap<String, Iterator<Xpp3Dom>>();
494 
495                     for ( Xpp3Dom recChild : recessive.childList )
496                     {
497                         if ( commonChildren.containsKey( recChild.name ) )
498                         {
499                             continue;
500                         }
501                         List<Xpp3Dom> dominantChildren = dominant.getChildrenAsList( recChild.name );
502                         if ( dominantChildren.size() > 0 )
503                         {
504                             commonChildren.put( recChild.name, dominantChildren.iterator() );
505                         }
506                     }
507 
508                     for ( int i = 0, recessiveChildCount = recessive.getChildCount(); i < recessiveChildCount; i++ )
509                     {
510                         Xpp3Dom recessiveChild = recessive.getChild( i );
511                         Iterator<Xpp3Dom> it = commonChildren.get( recessiveChild.getName() );
512                         if ( it == null )
513                         {
514                             dominant.addChild( new Xpp3Dom( recessiveChild ) );
515                         }
516                         else if ( it.hasNext() )
517                         {
518                             Xpp3Dom dominantChild = it.next();
519 
520                             String dominantChildCombinationMode =
521                                 dominantChild.getAttribute( SELF_COMBINATION_MODE_ATTRIBUTE );
522                             if ( SELF_COMBINATION_REMOVE.equals( dominantChildCombinationMode ) )
523                             {
524                                 dominant.removeChild( dominantChild );
525                             }
526                             else
527                             {
528                                 mergeIntoXpp3Dom( dominantChild, recessiveChild, childMergeOverride );
529                             }
530                         }
531                     }
532                 }
533             }
534         }
535     }

537     /**
538      * Merge two DOMs, with one having dominance in the case of collision.
539      *
540      * @see #CHILDREN_COMBINATION_MODE_ATTRIBUTE
541      * @see #SELF_COMBINATION_MODE_ATTRIBUTE
542      * @param dominant The dominant DOM into which the recessive value/attributes/children will be merged
543      * @param recessive The recessive DOM, which will be merged into the dominant DOM
544      * @param childMergeOverride Overrides attribute flags to force merging or appending of child elements into the
545      *            dominant DOM
546      * @return merged DOM
547      */
548     public static Xpp3Dom mergeXpp3Dom( Xpp3Dom dominant, Xpp3Dom recessive, Boolean childMergeOverride )
549     {
550         if ( dominant != null )
551         {
552             mergeIntoXpp3Dom( dominant, recessive, childMergeOverride );
553             return dominant;
554         }
555         return recessive;
556     }

558     /**
559      * Merge two DOMs, with one having dominance in the case of collision. Merge mechanisms (vs. override for nodes, or
560      * vs. append for children) is determined by attributes of the dominant root node.
561      *
562      * @see #CHILDREN_COMBINATION_MODE_ATTRIBUTE
563      * @see #SELF_COMBINATION_MODE_ATTRIBUTE
564      * @param dominant The dominant DOM into which the recessive value/attributes/children will be merged
565      * @param recessive The recessive DOM, which will be merged into the dominant DOM
566      * @return merged DOM
567      */
568     public static Xpp3Dom mergeXpp3Dom( Xpp3Dom dominant, Xpp3Dom recessive )
569     {
570         if ( dominant != null )
571         {
572             mergeIntoXpp3Dom( dominant, recessive, null );
573             return dominant;
574         }
575         return recessive;
576     }

582     @Override
583     public boolean equals( Object obj )
584     {
585         if ( obj == this )
586         {
587             return true;
588         }
589 
590         if ( !( obj instanceof Xpp3Dom ) )
591         {
592             return false;
593         }
594 
595         Xpp3Dom dom = (Xpp3Dom) obj;
596 
597         if ( name == null ? dom.name != null : !name.equals( dom.name ) )
598         {
599             return false;
600         }
601         else if ( value == null ? dom.value != null : !value.equals( dom.value ) )
602         {
603             return false;
604         }
605         else if ( attributes == null ? dom.attributes != null : !attributes.equals( dom.attributes ) )
606         {
607             return false;
608         }
609         else if ( childList == null ? dom.childList != null : !childList.equals( dom.childList ) )
610         {
611             return false;
612         }
613         else
614         {
615             return true;
616         }
617     }

625         result = 37 * result + ( attributes != null ? attributes.hashCode() : 0 );
626         result = 37 * result + ( childList != null ? childList.hashCode() : 0 );
627         return result;
